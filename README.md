# Set up
1. Install python on machine
   - brew install python
   - python --version
   - python3 --verison
2. Navigate to root project directory
3. Set up virtual environment
   - python3 -m venv .venv
4. Activate virtual environment
   - source .venv/bin/activate
5. Install requirements
   - pip3 install -r requirements.txt
